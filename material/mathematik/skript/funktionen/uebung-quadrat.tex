\subsection{"Ubungen zu quadratischen Funktionen}

\begin{enumerate}
\item Berechne die Scheitelpunkte folgender Gleichungen
\begin{itemize}
\item $f(x) = 3x^2 - 3x + 6$
\item $f(x) = x^2 - 8x - 4$
\item $f(x) = x^2 + 4x + 1$
\item $f(x) = -4x^2 + 16x - 1$
\end{itemize}
\item Berechne die Nullstellen der oberen Gleichungen
\end{enumerate}
